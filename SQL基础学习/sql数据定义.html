🏗️ 数据定义语言 (DDL)

CREATE / ALTER / DROP / TRUNCATE · 表、列、约束

CREATE 创建表

CREATE TABLE 表名 (
    列1 数据类型 [约束],
    列2 数据类型,
    PRIMARY KEY (列1),
    FOREIGN KEY (列2) REFERENCES 其他表(列)
);
-- 示例：学生表
CREATE TABLE students (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(50) NOT NULL,
    age INT CHECK (age > 0),
    enrolled_date DATE
);

ALTER 修改表

-- 添加列
ALTER TABLE 表名 ADD 列名 数据类型;

-- 修改列类型 (MySQL)
ALTER TABLE 表名 MODIFY 列名 新类型;

-- 删除列
ALTER TABLE 表名 DROP COLUMN 列名;

-- 重命名表
ALTER TABLE 旧表名 RENAME TO 新表名;

DROP 删除

DROP TABLE 表名;        -- 删除表结构和数据
DROP DATABASE 数据库名;  -- 删除数据库

TRUNCATE 清空表

TRUNCATE TABLE 表名;   -- 快速删除所有行，不可回滚
