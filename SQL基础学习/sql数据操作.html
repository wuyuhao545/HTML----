✏️ 数据操作语言 (DML)

INSERT · UPDATE · DELETE · 事务基本概念

INSERT 插入

-- 插入完整行
INSERT INTO 表名 VALUES (值1, 值2, ...);

-- 指定列
INSERT INTO 表名 (列1, 列2) VALUES (值1, 值2);

-- 批量插入
INSERT INTO 表名 (列1, 列2) VALUES 
    (值1, 值2),
    (值3, 值4);

UPDATE 更新

UPDATE 表名
SET 列1 = 新值1, 列2 = 新值2
WHERE 条件;   -- 务必加 WHERE，否则全表更新
-- 示例：部门5员工工资涨10%
UPDATE employees
SET salary = salary * 1.1
WHERE dept_id = 5;

DELETE 删除

DELETE FROM 表名 WHERE 条件;   -- 不加条件会清空表

事务控制（基本）

BEGIN;  -- 或 START TRANSACTION;
UPDATE ...;
DELETE ...;
COMMIT;    -- 提交
ROLLBACK;  -- 回滚
