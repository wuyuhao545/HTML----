🔍 数据查询语言 (DQL)

核心语句：SELECT · 过滤 · 分组 · 排序

基本 SELECT

-- 检索所有列
SELECT * FROM 表名;

-- 特定列 + 别名
SELECT 列1 AS 别名1, 列2 别名2 FROM 表名;

-- 去重
SELECT DISTINCT 列名 FROM 表名;

WHERE 过滤

SELECT 列1, 列2
FROM 表名
WHERE 条件;

-- 示例：工资大于6000 且 部门为 2 或 3
SELECT name, salary
FROM employees
WHERE salary > 6000 AND dept_id IN (2, 3);

聚合函数

COUNT(*), SUM(列), AVG(列), MAX(列), MIN(列)
-- 统计员工总数、平均工资
SELECT COUNT(*) AS 员工数, AVG(salary) AS 平均工资 FROM employees;

GROUP BY / HAVING

SELECT 列A, 聚合函数(列B)
FROM 表名
GROUP BY 列A
HAVING 分组后条件;   -- 例如平均工资 > 8000

ORDER BY 排序

SELECT ... FROM ...
ORDER BY 列1 DESC, 列2 ASC;
