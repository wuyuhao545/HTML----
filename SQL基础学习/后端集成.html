Python · Node.js · Java 连接数据库、执行查询/更新、使用参数化查询防止注入

🔷 Python (以 pymysql + sqlite3 为例)

连接与查询

import pymysql

# 建立连接
conn = pymysql.connect(
    host='localhost',
    user='root',
    password='123456',
    database='test',
    charset='utf8mb4'
)

try:
    with conn.cursor() as cursor:
        # 执行查询 (参数化查询，避免注入)
        sql = "SELECT id, name, salary FROM employees WHERE dept_id = %s"
        cursor.execute(sql, (5,))
        results = cursor.fetchall()
        for row in results:
            print(row)

        # 执行插入
        insert_sql = "INSERT INTO employees (name, salary, dept_id) VALUES (%s, %s, %s)"
        cursor.execute(insert_sql, ('王小明', 7500, 3))
        conn.commit()  # 提交事务
        print("插入成功，影响行数:", cursor.rowcount)
finally:
    conn.close()

使用 sqlite3 (内置)

import sqlite3

🔷 Node.js (使用 mysql2 或 pg)

const mysql = require('mysql2/promise');

async function main() {
    const conn = await mysql.createConnection({
        host: 'localhost',
        user: 'root',
        password: '123456',
        database: 'test'
    });

    // 参数化查询 (用 ? 占位)
    const [rows] = await conn.execute(
        'SELECT id, name, salary FROM employees WHERE dept_id = ?',
        [5]
    );
    console.log(rows);

    // 插入并获取影响行数
    const [result] = await conn.execute(
        'INSERT INTO employees (name, salary, dept_id) VALUES (?, ?, ?)',
        ['李芳', 8200, 4]
    );
    console.log('插入行数:', result.affectedRows);

    await conn.end();
}
main();

🔷 Java (JDBC)

import java.sql.*;

public class JdbcExample {
    public static void main(String[] args) {
        String url = "jdbc:mysql://localhost:3306/test";
        String user = "root";
        String password = "123456";

        try (Connection conn = DriverManager.getConnection(url, user, password)) {
            // 参数化查询
            String sql = "SELECT id, name, salary FROM employees WHERE dept_id = ?";
            PreparedStatement pstmt = conn.prepareStatement(sql);
            pstmt.setInt(1, 5);
            ResultSet rs = pstmt.executeQuery();
            while (rs.next()) {
                System.out.println(rs.getInt("id") + ", " + rs.getString("name"));
            }

            // 插入
            String insertSql = "INSERT INTO employees (name, salary, dept_id) VALUES (?, ?, ?)";
            PreparedStatement insertStmt = conn.prepareStatement(insertSql);
            insertStmt.setString(1, "张伟");
            insertStmt.setDouble(2, 9000);
            insertStmt.setInt(3, 2);
            int affected = insertStmt.executeUpdate();
            System.out.println("插入行数: " + affected);
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
}
⚠️ 重要安全实践：永远使用参数化查询（预处理语句），避免拼接 SQL 字符串，防止注入攻击。

🌐 通用步骤总结

步骤说明
1. 引入驱动/库根据语言和数据库选择对应驱动（如 pymysql、mysql2、JDBC）
2. 建立连接提供主机、端口、用户名、密码、数据库名
3. 创建语句对象通常使用 cursor 或 preparedStatement
4. 执行SQL传递 SQL 字符串和参数（占位符方式）
5. 处理结果遍历结果集或获取影响行数
6. 关闭连接释放资源（或使用 try-with-resources / with 语句自动关闭）
