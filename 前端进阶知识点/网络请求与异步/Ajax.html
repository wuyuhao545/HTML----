📦 Ajax 封装 · 回调式网络请求

通用 ajax 函数 · 回调处理 · 并发请求示例

ajax 函数封装GET / POST
function ajax({ method, url, data, success, error }) {
  const xhr = new XMLHttpRequest();
  xhr.open(method, url);
  if (method === 'POST') xhr.setRequestHeader('Content-Type', 'application/json');
  xhr.onload = () => {
    if (xhr.status >= 200 && xhr.status < 300) success(JSON.parse(xhr.responseText));
    else error(xhr.status);
  };
  xhr.onerror = () => error('网络错误');
  xhr.send(data ? JSON.stringify(data) : null);
}
📦 统一封装 XHR，简化调用。
POST 请求发送数据
ajax({
  method: 'POST',
  url: 'https://jsonplaceholder.typicode.com/posts',
  data: { title: 'Ajax封装', body: '测试', userId: 1 },
  success: res => console.log(res),
  error: err => console.error(err)
});
📨 使用封装后的 POST 请求。
并发请求（回调嵌套）模拟并发
// 通过嵌套回调实现并发
ajax({ url: '/posts/1', success: post => {
  ajax({ url: '/users/1', success: user => {
    console.log(post, user);
  }});
}});
⚡ 传统方式下并发请求会导致嵌套加深。
💡 Ajax 封装使得代码复用性更高，但回调方式仍难以应对复杂场景。
