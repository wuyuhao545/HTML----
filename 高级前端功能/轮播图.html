轮播图 1：自动轮播 (带悬停暂停)

▶ 自动播放

📌 轮播图1 核心代码 (自动轮播)

<script>
    // 轮播图1 - 自动轮播实现
    const images1 = [
        'https://picsum.photos/id/1015/800/400',
        'https://picsum.photos/id/1018/800/400',
        'https://picsum.photos/id/1043/800/400'
    ];
    let currentIndex1 = 0;
    const img1 = document.getElementById('carousel1Img');
    const prev1 = document.getElementById('prevBtn1');
    const next1 = document.getElementById('nextBtn1');
    const indicators1 = document.getElementById('indicators1');
    const hint1 = document.getElementById('autoPlayHint1');
    let autoInterval;
    let isPaused = false;

    function initIndicators1() {
        indicators1.innerHTML = '';
        images1.forEach((_, i) => {
            const dot = document.createElement('span');
            dot.className = 'indicator';
            dot.dataset.index = i;
            indicators1.appendChild(dot);
        });
    }
    function updateCarousel1() {
        img1.src = images1[currentIndex1];
        document.querySelectorAll('#indicators1 .indicator').forEach((dot, i) => {
            dot.classList.toggle('active', i === currentIndex1);
        });
    }
    function goToSlide1(index) {
        if (index < 0) index = images1.length - 1;
        else if (index >= images1.length) index = 0;
        currentIndex1 = index;
        updateCarousel1();
    }
    function nextSlide1() { goToSlide1(currentIndex1 + 1); }
    function prevSlide1() { goToSlide1(currentIndex1 - 1); }

    function startAutoPlay() {
        if (autoInterval) clearInterval(autoInterval);
        autoInterval = setInterval(() => {
            if (!isPaused) nextSlide1();
        }, 3000);
        hint1.textContent = '▶ 自动播放';
    }
    function stopAutoPlay() {
        if (autoInterval) {
            clearInterval(autoInterval);
            autoInterval = null;
            hint1.textContent = '⏸ 暂停';
        }
    }
    prev1.addEventListener('click', () => { prevSlide1(); stopAutoPlay(); startAutoPlay(); });
    next1.addEventListener('click', () => { nextSlide1(); stopAutoPlay(); startAutoPlay(); });

    document.getElementById('carousel1').addEventListener('mouseenter', () => isPaused = true);
    document.getElementById('carousel1').addEventListener('mouseleave', () => isPaused = false);

    indicators1.addEventListener('click', (e) => {
        if (e.target.classList.contains('indicator')) {
            const idx = parseInt(e.target.dataset.index, 10);
            goToSlide1(idx);
            stopAutoPlay();
            startAutoPlay();
        }
    });

    initIndicators1();
    updateCarousel1();
    startAutoPlay();
</script>
⭐ 注释：自动轮播图实现了图片切换、指示点、自动播放（3秒间隔）、鼠标悬停暂停、点击指示点跳转。点击左右按钮会重置自动播放计时器。
